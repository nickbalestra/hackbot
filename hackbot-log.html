    Log script

    Print on screen users messages.

    Listener

        robot.hear(/(.)/i, function(msg) {
        var text = msg.message.text;
        var user = msg.message.user;
        var date = msg.message.createdAt;

        return msg.robot.adapter.print(user.name + ' said: ' + text + ' ( ' + date + ' )' );
    });
});
